Задача 2 — Подсказка

Console.log

Пример использования команды console.log()

alert('Вам будет предложено ввести 2 числа');


let answer = confirm('Хотите узнать их сумму?');

let a, b;


if (answer == true) {

a = +prompt('число 1?');

b = +prompt('число 2?');

}


const result = a + b;


if (!isFinite(result)) {

alert('Вы отменили ввод или указали не число!');

} else {

alert('Результат вы узнаете в консоли :)');

console.log(result);

}

Prompt

Пример использования команды prompt()

let birthYear = +prompt('Год вашего рождения?', '');

const currentYear = 2022;

const yourAge = currentYear - birthYear;


if (!isFinite(birthYear)) {

alert('Вы указали не число');

} else if (birthYear > 2022) {

alert('Вы из будущего!');

} else if (birthYear < 1922) {

alert('Ого! Вот это вы долгожитель!');

} else {

alert(`Вам ${yourAge} лет(год)`);

}
